## 修改 2: 修复公共资源库文件保存时 IndexedDB 缓存路径错误

### 问题描述

用户发现点击保存时,IndexedDB 中出现了 `/api/library/block/filesData/cmnsaru53000d4sufzcg9wjlg` 这个记录,而且:

1. 路径不正确:应该包含完整的年月目录和文件扩展名,如 `202604/cmnsaru53000d4sufzcg9wjlg.dwg.mxweb`
2. 数据类型不对:存储的是 `Uint8Array`,而其他记录都是 `ArrayBuffer`
3. 缺少时间戳参数:应该与打开文件时的 URL 一致,包含 `?t=timestamp`

**正确的 IndexedDB key 应该是**:

```
/api/library/block/filesData/202604/cmnsaru53000d4sufzcg9wjlg.dwg.mxweb?t=1775789034887
```

### 根本原因分析

1. **Mx_Save 命令缺少公共资源库文件的保存逻辑**:当 `currentFileInfo.libraryKey` 存在时(表示是公共资源库文件),没有特殊处理,导致走到错误的分支
2. **缺少 `path` 字段**:`setCurrentFileInfo` 时没有保存节点的完整路径 (`node.path`)
3. **保存时使用 fileId 而不是 path**:构建 IndexedDB key 时使用了 `fileId`,而不是完整的 `nodePath`
4. **数据类型未转换**:`MxCpp.App.getCurrentMxCAD().saveFile()` 回调返回的 `data` 是 `Uint8Array`,直接存入 IndexedDB,没有转换为 `ArrayBuffer`
5. **缺少时间戳**:保存时没有添加 `?t=timestamp` 参数,与打开文件时的 URL 不一致

`packages/frontend/src/services/mxcadManager.ts`

#### 2.5 新增 `saveLibraryFile` 函数

**位置**: 约 1790-1914 行

**完整实现**:

```typescript
/**
 * 保存公共资源库文件(图纸库/图块库)
 */
async function saveLibraryFile() {
  const { fileId, name, libraryKey, path: nodePath } = currentFileInfo!;

  if (!libraryKey) {
    globalShowToast('保存失败:未知的资源库类型', 'error');
    return;
  }

  if (!nodePath) {
    console.warn('[saveLibraryFile] 缺少 nodePath,无法更新 IndexedDB 缓存');
    globalShowToast('保存失败:缺少文件路径信息', 'error');
    return;
  }

  showLoadingOverlay('正在保存文件...');

  const savedFile = await new Promise<{
    blob: Blob;
    data: ArrayBuffer | Uint8Array;
    filename: string;
  }>((resolve, reject) => {
    MxCpp.App.getCurrentMxCAD().saveFile(
      name,
      (data) => {
        try {
          let blob: Blob;
          const isSafari = /^((?!chrome|android).)*safari/i.test(
            navigator.userAgent
          );
          if (isSafari) {
            blob = new Blob([data.buffer], {
              type: 'application/octet-stream',
            });
          } else {
            blob = new Blob([data.buffer], {
              type: 'application/octet-binary',
            });
          }
          resolve({
            blob,
            data,
            filename: name,
          });
        } catch (e) {
          reject(e);
          console.error('保存文件失败', e);
        }
      },
      false,
      false,
      undefined
    );
  });

  try {
    const { libraryApi } = await import('../services/libraryApi');
    const isDrawing = libraryKey === 'drawing';

    updateLoadingMessage('正在上传到服务器...');

    // 上传到服务器
    if (isDrawing) {
      await libraryApi.uploadDrawingChunk(
        createUploadFormData(savedFile.blob, fileId, name)
      );
    } else {
      await libraryApi.uploadBlockChunk(
        createUploadFormData(savedFile.blob, fileId, name)
      );
    }

    // ✅ 更新本地缓存 - 使用 node.path 构建正确的 URL
    const fileUrl = isDrawing
      ? `/api/library/drawing/filesData/${nodePath}`
      : `/api/library/block/filesData/${nodePath}`;

    // ✅ 添加时间戳参数(与打开文件时的 URL 保持一致)
    const updatedAt = await getNodeUpdatedAt(fileId, libraryKey);
    const timestamp = updatedAt ? new Date(updatedAt).getTime() : Date.now();
    const fileUrlWithTimestamp = `${fileUrl}?t=${timestamp}`;

    console.log(`[saveLibraryFile] 更新 IndexedDB 缓存`);
    console.log(`  - fileId: ${fileId}`);
    console.log(`  - nodePath: ${nodePath}`);
    console.log(`  - fileUrl: ${fileUrlWithTimestamp}`);
    console.log(`  - 原始数据类型: ${savedFile.data.constructor.name}`);

    const db = await new Promise<IDBDatabase>((resolve, reject) => {
      const request = indexedDB.open('emscripten_filesystem', 1);
      request.onerror = () => reject(request.error);
      request.onsuccess = () => resolve(request.result);
    });

    const transaction = db.transaction(['FILES'], 'readwrite');
    const objectStore = transaction.objectStore('FILES');

    // ✅ 确保存入的是 ArrayBuffer 而不是 Uint8Array
    const arrayBufferData =
      savedFile.data instanceof Uint8Array
        ? savedFile.data.buffer
        : savedFile.data;

    console.log(`  - 转换后数据类型: ${arrayBufferData.constructor.name}`);

    await new Promise<void>((resolve, reject) => {
      const request = objectStore.put(arrayBufferData, fileUrlWithTimestamp);
      request.onerror = () => reject(request.error);
      request.onsuccess = () => resolve();
    });

    console.log(`[saveLibraryFile] IndexedDB 缓存更新成功`);

    resetDocumentModified();
    hideLoadingOverlay();
    globalShowToast('文件保存成功', 'success');
  } catch (error) {
    console.error('保存公共资源库文件失败:', error);
    hideLoadingOverlay();
    globalShowToast(
      error instanceof Error ? error.message : '保存失败,请稍后重试',
      'error'
    );
  }
}
```

#### 2.6 新增辅助函数 `createUploadFormData`

**位置**: 约 1919-1929 行

```typescript
/**
 * 创建上传表单数据
 */
function createUploadFormData(
  blob: Blob,
  nodeId: string,
  filename: string
): FormData {
  const formData = new FormData();
  formData.append('file', blob, filename);
  formData.append('nodeId', nodeId);
  formData.append(
    'hash',
    `${Date.now()}_${Math.random().toString(36).substring(7)}`
  );
  formData.append('name', filename);
  formData.append('size', String(blob.size));
  formData.append('chunk', '0');
  formData.append('chunks', '1');
  return formData;
}
```

#### 2.7 新增辅助函数 `getNodeUpdatedAt`

**位置**: 约 1934-1950 行

```typescript
/**
 * 获取节点的 updatedAt 时间戳
 */
async function getNodeUpdatedAt(
  nodeId: string,
  libraryKey: string
): Promise<string | null> {
  try {
    const { libraryApi } = await import('../services/libraryApi');
    const response =
      libraryKey === 'drawing'
        ? await libraryApi.getDrawingNode(nodeId)
        : await libraryApi.getBlockNode(nodeId);

    const node = response.data as any;
    return node?.updatedAt || null;
  } catch (error) {
    console.warn('[getNodeUpdatedAt] 获取失败:', error);
    return null;
  }
}
```
